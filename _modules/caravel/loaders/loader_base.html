        """ A method checking the file extension.

        Parameters
        ----------
        path: str
            the path to the data to be loaded.

        Returns
        -------
        out: bool
            True if the file extension is valid, False otherwise.
        """
        for ext in self.allowed_extensions:
            if path.endswith(ext):
                return True
        return False
